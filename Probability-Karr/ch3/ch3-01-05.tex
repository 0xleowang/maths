\begin{exercise}
  Let $A_1,\ldots,A^n$ be independent events. Prove that $P(\cup_{i=1}^nA_i)=1-\Pi_{i=1}^n[1-P(A_i)]$.
\end{exercise}
\begin{solution}
  \begin{align*}
    P(\cup_{i=1}^nA_i) &= 1 - P(\cap_{i=1}^n A_i^c) \\
      &= 1 - \Pi_{i=1}^n P(A_i^c)  \quad(\mbox{by independence}) \\
      &= 1 - \Pi_{i=1}^n (1-P(A_i))
  \end{align*}
\end{solution}
